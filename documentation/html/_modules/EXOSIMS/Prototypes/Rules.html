from scipy import special as sp
import numpy as np
from astropy import units as u


[docs]class Rules(object):
    """Rules class template
    
    This class contains all variables and methods necessary to perform
    Rules Module calculations in exoplanet mission simulation"""

    modtype = 'Rules'
    
    def __init__(self, **specs):
        """Initializes Rules object with default values or values from user
        specification
        
        Input: specs - dictionary containing user specification values"""
        
        # default values
        self.FAP = 0.01/1000. # false alarm probability
        self.MDP = 0.001 # missed detection probability
        self.dutyCycle = 1. # Observation duty cycle
        self.intCutoff = 50.*u.day # integration cutoff
        
        # replace default values with any user specified values
        atts = self.__dict__.keys()
        for att in atts:
            if att in specs:
                if att == 'intCutoff':
                    setattr(self, att, specs[att]*u.day)
                else:
                    setattr(self, att, specs[att])
                    
        # initialize values updated by functions
        
        # set values derived from quantities above
        self.K = self.find_K() # threshold value determined by FAP
        self.gamma = self.find_gamma() # threshold value determined by MDP
                

[docs]    def find_K(self):
        """Finds threshold value determined by FAP"""
        return np.sqrt(2.)*sp.erfinv(2.*(1.-self.FAP)-1.)
        


[docs]    def find_gamma(self):
        """Finds threshold value determined by MDP"""
        return np.sqrt(2.)*sp.erfinv(2.*self.MDP-1.)
