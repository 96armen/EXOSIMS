[docs]class OpticalSys(object):
    """Optical System class template
    
    This class contains all variables and methods necessary to perform
    Optical System Definition Module calculations in exoplanet mission 
    simulation"""

    modtype = 'OpticalSys'
    
    def __init__(self, **specs):
        """Initializes Optical System with default values or values from user
        specification
        
        Input: specs - dictionary containing user specification values"""
        
        # default Optical System values
        self.lam = 500.*u.nm # detection wavelength (nm)
        self.shapeFac = np.pi/4. # shapeFac*diameter**2 = Area
        self.pupilArea = 4.*np.pi*(u.m)**2 # entrance pupil area
        self.SNchar = 11. # Signal to Noise Ratio for characterization
        self.haveOcculter = False # boolean signifying if system has occulter
        self.pixelArea = 1e-10*(u.m)**2 # pixel area (m**2)
        self.focalLength = 240.*u.m # focal length (m)
        self.IWA = 0.075*u.arcsec # Inner Working Angle (arcseconds)
        self.OWA = np.inf*u.arcsec # Outer Working Angle (arcseconds)
        self.dMagLim = 26. # limiting delta magnitude
        self.throughput = 0.5 # optical system throughput at IWA
        self.contrast = 1.e-10 # optical system designed suppression level at IWA
        self.dr = 0.001*(1/u.s) # detector dark-current rate per pixel
        self.sigma_r = 3. # detector read noise (e/s)
        self.t_exp = 1000.*u.s # exposure time per read (s)
        self.QE = 0.5 # detection quantum efficiency
        self.eta2 = 0.57 # post coronagraph attenuation
        self.deltaLambda = 100.*u.nm # bandpass (nm)
        self.have2ndDist = False # boolean signifying if there is a 2nd occulter
        # replace default values with any user specified values
        atts = self.__dict__.keys()
        for att in atts:
            if att in specs:
                if att == 'lam' or att == 'deltaLambda':
                    # set lam with proper units
                    setattr(self, att, specs[att]*u.nm)
                elif att == 'pupilArea' or att == 'pixelArea':
                    # set pupilArea or pixelArea with proper units
                    setattr(self, att, specs[att]*(u.m)**2)
                elif att == 'focalLength':
                    # set focalLength with proper units
                    setattr(self, att, specs[att]*u.m)
                elif att == 'IWA' or att == 'OWA':
                    # set IWA or OWA with proper units
                    setattr(self, att, specs[att]*u.arcsec)
                elif att == 'dr':
                    # set dr with proper units
                    setattr(self, att, specs[att]*(1/u.s))
                elif att == 't_exp':
                    # set t_exp with proper units
                    setattr(self, att, specs[att]*u.s)
                else:
                    setattr(self, att, specs[att])
                    
        # set values derived from quantities above
        if np.isinf(self.OWA.value):
            self.OWA = np.pi/2

[docs]    def calc_maxintTime(self, targlist):
        """Finds maximum integration time for target systems (called in
        TargetList)
        
        Input: targlist - TargetList module object containing target list star 
        attributes.
        targlist also contains the following available for use:
        targlist.opt - optical system object
        targlist.pop - planet population object
        targlist.rules - rules object
        targlist.zodi - zodiacal light object
        targlist.comp - completeness object
        
        Output: maxintTime - numpy array of maximum integration times
        
        This defines the data type expected, specific OpticalSys classes give
        maximum integration times."""
        
        # maximum integration time is given as 1 day for each system in target list
        maxintTime = np.array([1.]*len(targlist.Name))*u.day
        
        return maxintTime

        """Finds integration time for a specific target system (called in
        SurveySimulation)
        
        Input: targlist - TargetList module object 
        opt - OpticalSys module object
        rules - Rules module object
        universe - SimulatedUniverse module object
        s_ind - target star index
        planInds - planet index linking back to target star
        Output: intTime - numpy array of integration times
        
        This defines the data type expected, specific OpticalSys classes give
        integration times."""
        
        # integration time given as 1 day
        intTime = np.array([1.]*len(planInds))*u.day
        
        return intTime
