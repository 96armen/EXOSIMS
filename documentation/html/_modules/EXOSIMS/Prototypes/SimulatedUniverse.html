    """Simulated Universe class template
    
    This class contains all variables and functions necessary to perform 
    Simulated Universe Module calculations in exoplanet mission simulation
    
    It inherits the following class objects which are definied in __init__:
    TL - TargetList
    PlanPhys - PlanetPhysical"""

    modtype = 'SimulatedUniverse'
    
    def __init__(self, **specs):
        """Initializes simulated universe
        
        Input: specs - dictionary containing user specification values and 
        desired module names"""
        
        # get desired module names (prototype or specific)
        # initialize names to prototype names
        targlistname = 'TargetList'
        planphysname = 'PlanetPhysical'
        
        names = ['targlistname', 'planphysname']
        # rewrite if named in specs
        for name in names:
            if name in specs:
                if name == 'targlistname':
                    targlistname = specs[name]
                elif name == 'planphysname':
                    planphysname = specs[name]
        
        # import TargetList class
        TL = self.get_module(targlistname, 'TargetList')
        # import PlanetPhysical class
        PlanPhys = self.get_module(planphysname, 'PlanetPhysical')
        
        self.targlist = TL(**specs)
        self.planphys = PlanPhys(**specs)
        
        # bring inherited class objects to top level of Simulated Universe
        self.opt = self.targlist.opt # optical system class object
        del self.targlist.opt
        self.pop = self.targlist.pop # planet population class object
        del self.targlist.pop
        self.rules = self.targlist.rules # rules class object
        del self.targlist.rules
        self.zodi = self.targlist.zodi # zodiacal light class object
        del self.targlist.zodi
        self.comp = self.targlist.comp # completeness class object
        del self.targlist.comp
        
        # planets mapped to target stars
        self.planInds = self.planet_to_star()
        self.nPlans = len(self.planInds) # number of planets
        # indices of target stars with planets
        self.sysInds = np.unique(self.planInds)
        self.Mp = self.planet_masses() # planet masses
        self.Rp = self.planet_radii() # planet radii
        self.r = self.planet_pos() # planet initial positions
        self.v = self.planet_vel() # planet intial velocities
        self.p = self.planet_albedos() # planet albedos
        # inclination in degrees of each planet
        self.I = self.planet_inclinations()
        # rock/ice fraction for rocky/icy planets
        self.rtype = self.planet_rtypes()
        # exozodi levels for systems with planets
        self.fzodicurr = self.zodi.fzodi(self)

    def __str__(self):
        """String representation of Simulated Universe object
        
        When the command 'print' is used on the Simulated Universe object,

        atts = self.__dict__.keys()
        
        for att in atts:

[docs]    def planet_to_star(self):
        """Assigns planets to stars and returns index of star in target star
        list
        
        This method has access to the following:
        self.targlist - target list object
        self.planphys - planet physical model object
        self.opt - optical system object
        self.pop - planet population object
        self.rules - rules object
        self.zodi - zodiacal light object
        self.comp - completeness object
        
        Output: planSys - numpy array containing indices of the target star 
        each planet (element in the list) belongs to
        
        This defines the data type expected, specific SimulatedUniverse class
        objects will populate these indices more realistically"""
        
        # assign one planet to each star in the target list
        planSys = np.where(~np.isnan(self.targlist.parx))[0]
        
        return planSys

[docs]    def prop_system(self, r, v, Mp, Ms, dt):
        """Propagates planets' state vectors (position and velocity) forward
        in time using the Kepler state transition matrix
        
        Input: r - numpy array containing planet position vectors relative to
        host stars (astropy unit attached)
        v - numpy array containing planet velocity vectors relative to host
        stars (astropy unit attached)
        Mp - numpy array containing list of planet masses (astropy unit attached)
        Ms - numpy array containing mass of target star (astropy unit attached)
        dt - time increment to propagate the system (astropy unit attached)
        
        Output: rnew - propagated position vectors (astropy unit km)
        vnew - propagated velocity vectors (astropy unit km/day)"""
        
        # stack dimensionless positions and velocities
        x0 = np.array([])
        for i in xrange(len(r)):
            x0 = np.hstack((x0, r[i].to(u.km).value, v[i].to(u.km/u.day).value))
        # calculate vector of gravitational parameter    
        mu = (const.G*(Mp + Ms*const.M_sun)).to(u.km**3/u.day**2).value
        if np.isscalar(mu):
            mu = np.array([mu])
        # use keplerSTM.py to propagate the system
        prop = planSys(x0, mu)
        prop.takeStep(dt.to(u.day).value)
        # split off position and velocity vectors
        x1 = np.array(np.hsplit(prop.x0, 2*len(r)))
        rind, vind = [], []
        for x in xrange(len(x1)):
            if x%2 == 0:
                rind.append(x)
            else:
                vind.append(x)
        rind, vind = np.array(rind), np.array(vind)
        
        # assign new position and velocity arrays
        rnew = x1[rind]*u.km
        vnew = x1[vind]*u.km/u.day
        
        return rnew, vnew

        """Imports specific or Prototype class module
        
        Input: name - string containing module name
        folder - specific module folder
        Output: desired class module"""
                
        try:
            package = 'EXOSIMS.'+folder+'.'+name
            b = importlib.import_module(package)
            print 'Imported specific %s module %s' % (folder, name)
            desiredmodule = getattr(b, name)
        except ImportError:
            package = 'EXOSIMS.Prototypes.'+folder
            b = importlib.import_module(package)
            print 'Imported %s prototype module' % folder
            desiredmodule = getattr(b, folder)
        
        return desiredmodule
