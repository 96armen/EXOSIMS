    def __init__(self, **specs):
        """Initializes Survey Simulation with default values
        
        Input: specs - dictionary containing user specification values and 
        desired module names"""
        
        # get desired module names (prototype or specific)
        # initialize names to prototype names
        simuniname = 'SimulatedUniverse'
        obsname = 'Observatory'
        timename = 'TimeKeeping'
        ppname = 'PostProcessing'
        
        names = ['simuniname', 'obsname', 'timename', 'ppname']
        # rewrite if named in specs
        for name in names:
            if name in specs:
                if name == 'simuniname':
                    simuniname = specs[name]
                elif name == 'obsname':
                    obsname = specs[name]
                elif name == 'timename':
                    timename = specs[name]
                elif name == 'ppname':
                    ppname = specs[name]
        
        # import simulated universe class
        SimUni = self.get_module(simuniname, 'SimulatedUniverse')
        # import observatory class
        Obs = self.get_module(obsname, 'Observatory')
        # import timekeeping class
        TK = self.get_module(timename, 'TimeKeeping')
        # import postprocessing class
        PP = self.get_module(ppname, 'PostProcessing')
        
        self.simuni = SimUni(**specs)
        self.obs = Obs(**specs)
        self.timekeeping = TK(**specs)
        self.post = PP(**specs)
        
        # bring inherited class objects to top level of Survey Simulation
        self.opt = self.simuni.opt # optical system class object
        del self.simuni.opt
        self.rules = self.simuni.rules # rules class object
        del self.simuni.rules
        self.pop = self.simuni.pop # planet population class object
        del self.simuni.pop
        self.zodi = self.simuni.zodi # zodiacal light class object
        del self.simuni.zodi
        self.comp = self.simuni.comp # completeness class object
        del self.simuni.comp
        self.planphys = self.simuni.planphys # planet physical model class object
        del self.simuni.planphys
        self.targlist = self.simuni.targlist # target list class object
        del self.simuni.targlist
        
        # initialize values updated by functions
        # list of simulation results
        self.DRM = np.array([])
        # self.DRM contains in columns:
        # 0 - star index

        """Performs the survey simulation for missions with occulters and 
        populates self.DRM with results
        
        This method has access to the following:
        self.simuni - simulated universe object
        self.obs - observatory object
        self.timekeeping - time object
        self.post - post-processing object
        self.opt - optical system object
        self.rules - rules object
        self.pop - planet population object
        self.zodi - zodiacal light object
        self.comp - completeness object
        self.planphys - planet physical model object
        self.targlist - target list object"""
        
        # initialize values updated later
        # number of visits to target star
        visited = np.zeros((len(self.targlist.dist),), dtype=int)
        # target revisit list
        revisit_list = []
        # current time (normalized to zero at mission start) of planet positions
        planPosTime = np.array([self.timekeeping.currenttimeNorm.to(u.day).value]*self.simuni.nPlans)*u.day
        # number of planet observations
        observed = np.zeros((self.simuni.nPlans,), dtype=int)
        # set occulter separation
        self.obs.currentSep = self.obs.occulterSep
        
        # initialize run options
        # perform characterization if doChar = True
        if self.opt.SNchar > 0.:
            doChar = True
            # keep track of spectral characterizations, 0 is no characterization
            if self.opt.have2ndDist:
                spectra = np.zeros((self.simuni.nPlans,2), dtype=int)
            else:
                spectra = np.zeros((self.simuni.nPlans,1), dtype=int)
        else:
            doChar = False
            
        # get index of first target star
        s_ind = self.initial_target()
        
        # loop until mission is finished
        while self.timekeeping.currenttimeNorm < self.timekeeping.missionFinishNorm:
            # time at beginning of this iteration
            iter_start_time = self.timekeeping.currenttimeNorm
            # get target list star index of detections for extended_list
            if self.timekeeping.currenttimeNorm > self.timekeeping.missionLife and extended_list.shape[0] == 0:
                extended_list = np.unique(self.DRM[DRM[:,2]==1,0])
            
            # find planets belonging to target star
            pInds = np.where(self.simuni.planInds == s_ind)[0]
            
            # update visited list for current star
            visited[s_ind] += 1
            
            # in MATLAB find disturbance forces on occulter (genMission_v4
            # lines 116-122)

            # determine if there are any planets at the target star and
            # propagate the system if necessary
            if np.any(pInds):
                # there are planets so observation is possible
                observationPossible = np.ones((len(pInds),), dtype=bool)
                # if planet position times do not match up with current time
                # propagate the positions to the current time
                if planPosTime[pInds][0] != self.timekeeping.currenttimeNorm:
                    # propagate planet positions and velocities
                    self.simuni.r[pInds], self.simuni.v[pInds] = \
                    self.simuni.prop_system(self.simuni.r[pInds], 
                                        self.simuni.v[pInds], 
                                        self.simuni.Mp[pInds], 
                                        self.targlist.MsTrue[self.simuni.planInds[pInds]], 
                                        self.timekeeping.currenttimeNorm - planPosTime[pInds][0])
                    # update planet position times
                    planPosTime[pInds] = self.timekeeping.currenttimeNorm
            else:
                observationPossible = False
            print 'planets at target star'
            print observationPossible
            # determine if planets are inside IWA or outside OWA
            if np.any(observationPossible):
                observationPossible = self.check_IWA_OWA(observationPossible,self.simuni.r[pInds],self.targlist.dist[s_ind],self.opt.IWA,self.opt.OWA)
            print 'inside IWA'
            print observationPossible
            # determine if planets are bright enough
            if np.any(observationPossible):
                observationPossible = self.check_brightness(observationPossible, self.simuni.r[pInds], self.simuni.Rp[pInds], self.simuni.p[pInds], self.opt.dMagLim)
            print 'bright enough'
            print observationPossible
            # set integration time to max integration time as a default
            t_int = self.targlist.maxintTime[s_ind]
            
            # determine integration time and update observationPossible
            if np.any(observationPossible):
                t_trueint = self.opt.calc_intTime(self.targlist, self.opt, self.rules, self.simuni, s_ind, pInds)
                observationPossible = np.logical_and(observationPossible, t_trueint <= self.rules.intCutoff)
            print 'integration time'
            print observationPossible    
            # determine if planets are observable at the end of observation
            # and update integration time
            if np.any(observationPossible):
                t_int, observationPossible = self.check_visible_end(observationPossible, t_int, t_trueint, s_ind, pInds)
            print 'observable at end'
            print observationPossible
            
            t_int += self.obs.settling_time
            
            #------------------------------------------------------------------
            # OCCULTER ONLY
            # find disturbance forces on occulter
            dF_lateral, dF_axial = self.obs.distForces(self.timekeeping, self.targlist, s_ind, self.obs.currentSep == self.obs.occulterSep)
            # decrement mass for station-keeping
            mass_used, deltaV = self.obs.mass_dec(dF_lateral, t_int)
            self.obs.sc_mass -= mass_used
            #------------------------------------------------------------------
            
            # update mission times
            self.timekeeping.update_times(t_int)
            
            # temporary time stepping
            if visited[s_ind] == 3:
                self.timekeeping.currenttimeNorm = 100*u.year

            
            
        
        return 'run_Occulter_sim'

[docs]    def initial_target(self):
        """Returns index of initial target star
        
        This method has access to the following:
        self.simuni - simulated universe object
        self.obs - observatory object
        self.timekeeping - time object
        self.post - post-processing object
        self.opt - optical system object
        self.rules - rules object
        self.pop - planet population object
        self.zodi - zodiacal light object
        self.comp - completeness object
        self.planphys - planet physical model object
        self.targlist - target list object
        
        Output: s_ind - index of initial target star"""
        
        sinds = np.array([])
        dt = 1.*u.day
        while sinds.size == 0:
            a = self.obs.keepout(self.timekeeping.currenttimeAbs, self.targlist)
            # find observable targets at current mission time            
            sinds = np.where(self.obs.kogood)[0]
            # if no observable targets, update mission time by one day
            if sinds.size == 0:
                self.timekeeping.update_times(dt)
            # if the current mission time is greater than the mission lifetime
            # break this loop
            if self.timekeeping.currenttimeNorm > self.timekeeping.missionLife:
                break
        
        if self.timekeeping.currenttimeNorm > self.timekeeping.missionLife:
            print 'No targets available'
            
        s0 = self.targlist.comp0[sinds].argmax()
        s_ind = sinds[s0]
        
        return s_ind
        


[docs]    def check_IWA_OWA(self, observationPossible, r, dist, IWA, OWA):
        """Determines whether planets are inside the IWA or outside the OWA
        and returns numpy array of booleans where True is outside IWA and
        inside OWA
        
        Inputs: observationPossible - numpy array of booleans indicating if an
        observation of the planet is possible
        r - numpy array of planet position vectors
        dist - target star distance (in pc)
        IWA - instrument inner working angle
        OWA - instrument outer working angle
        
        Output: observationPossible"""
        
        # find apparent separation
        s = np.sqrt(r[:,0]**2 + r[:,1]**2)
        
        observationPossible = np.logical_and(observationPossible, np.logical_and(
        s >= dist*u.pc*np.tan(IWA), s <= dist*u.pc*np.tan(OWA)))
        
        return observationPossible
        


[docs]    def check_brightness(self, observationPossible, r, Rp, p, dMagLim):
        """Determines if planets are bright enough for detection and returns
        numpy array of booleans where True is a planet bright enough for
        detection
        
        Inputs: observationPossible - numpy array of booleans indicating if an
        observation of the planet is possible
        r - numpy array of planet position vectors
        Rp - numpy array of planet radii
        p - numpy array of planet albedos
        dMagLim - limiting difference in brightness between planet and star
        
        Output: observationPossible"""
        
        rmag = np.linalg.norm(r.to(u.AU).value, axis=1)
        # planetary phase function calculation, from Sobolev 1975
        beta = np.arccos(r[:,2].to(u.AU).value/rmag)
        Phi = (np.sin(beta) + (np.pi - beta)*np.cos(beta))/np.pi
        a = -2.5*np.log10(Rp.to(u.AU).value**2*p)
        b = -2.5*np.log10(Phi)
        c = 5.*np.log10(rmag)
        # delta magnitude of each planet
        dMag = a + b + c
        observationPossible = np.logical_and(observationPossible, dMag <= dMagLim)
        
        return observationPossible
    


[docs]    def check_visible_end(self, observationPossible, t_int, t_trueint, s_ind, pInds):
        """Determines if planets are visible at the end of the observation time
        and returns true integration time and observationPossible
        
        Inputs: observationPossible - numpy array of booleans indicating if an
        observation of the planet is possible
        t_int - integration time
        t_trueint - list of calculated integration times for planets
        s_ind - target star index
        pInds - numpy array of planet indices
        
        makes use of inherited objects:
        self.targlist - target list object 
        self.obs - observatory object 
        self.timekeeping - timekeeping object 
        self.simuni - simulated universe object 
        self.opt - optical system object 
        
        Outputs: t_int - true integration time for planetary system
        observationPossible"""
        
        for i in xrange(len(observationPossible)):
            if observationPossible[i]:
                r = self.simuni.r[pInds[i]]
                v = self.simuni.v[pInds[i]]
                Mp = self.simuni.Mp[pInds[i]]
                MsTrue = self.targlist.MsTrue[s_ind]
                Rp = self.simuni.Rp[pInds[i]]
                p = self.simuni.p[pInds[i]]
                dt = t_trueint[i] + self.obs.settling_time

                obsRes = self.sys_end_res(dt, s_ind, r, v, Mp, MsTrue, Rp, p)
                
                if obsRes == 1:
                    # planet visible at the end of observation
                    observationPossible[i] = True
                    # update integration time
                    if t_int == self.targlist.maxintTime[s_ind]:
                        t_int = t_trueint[i]
                    else:
                        if t_trueint[i] > t_int:
                            t_int = t_trueint[i]
        
        return t_int, observationPossible
        


[docs]    def sys_end_res(self, dt, s_ind, r, v, Mp, MsTrue, Rp, p):
        """Determines if a planet is visible at the end of the observation time
        and returns a value signifying this status
        
        Inputs: dt - propagation time
        s_ind - target star index
        r - planet position vector
        v - planet velocity vector
        Mp - planetary mass
        MsTrue - stellar mass
        Rp - planet radius
        p - planet albedo

        makes use of inherited objects:        
        self.obs - observatory object
        self.timekeeping - timekeeping object
        self.targlist - target list object
        self.opt - optical system object
        
        Output: obsRes - (-1) if star is inside the keepout zone, (0) if planet
        is not visible, (1) if the planet is visible"""
        
        # update keepout to end of observation time
        a = self.obs.keepout(self.timekeeping.currenttimeAbs+dt, self.targlist)
        # if star is not inside keepout zone at end of observation, see if 
        # planet is still observable
        if self.obs.kogood[s_ind]:
            # propagate planet to observational period end and find dMag
            rend, vend = self.simuni.prop_system(r, v, Mp, MsTrue, dt)
            # calculate dMagend
            send = np.sqrt(rend[0].to(u.km)**2 + rend[1].to(u.km)**2)
            rendmag = np.linalg.norm(rend.to(u.km).value)
            beta = np.arccos(rend[2].value/rendmag)
            Phi = (np.sin(beta) + (np.pi - beta)*np.cos(beta))/np.pi
            dMagend = -2.5*np.log10(Rp.to(u.AU).value**2*p) -2.5*np.log10(Phi) + 5.*np.log10(rendmag)
           
            if dMagend <= self.opt.dMagLim and send >= self.targlist.dist[s_ind]*u.pc*np.tan(self.opt.IWA):
                obsRes = 1
            else:
                obsRes = 0
        else:
            obsRes = -1
            
        return obsRes
