from EXOSIMS.Prototypes.StarCatalog import StarCatalog
import os, inspect


[docs]class SIMBADCatalog(StarCatalog):
    """SIMBAD Catalog class
    
    This class populates the star catalog used in EXOSIMS from the SIMBAD 
    catalog"""
    
    def __init__(self, filename):
        """Populates the EXOSIMS star catalog with data from the SIMBAD catalog
        
        Input: filename - input .pkl file or .mat file containing star catalog 
        data (no extension included in filename)"""
        
        classpath = os.path.split(inspect.getfile(self.__class__))[0]
        StarCatalog.__init__(self)
        pklpath = os.path.join(classpath, filename+'.pkl')
        # check if given filename exists as .pkl file already
        if os.path.exists(pklpath):
            self.populatepkl(filename+'.pkl')
        # check if given filename exists as a .mat file but not .pkl file
        elif os.path.exists(os.path.join('StarCatalog', filename+'.mat')):
            self.SIMBAD_mat2pkl(filename)
            self.populatepkl(filename+'.pkl')
        # otherwise load SIMBAD300.pkl
        else:
            self.populatepkl('SIMBAD300.pkl')
            print 'Loaded SIMBAD300.pkl star catalog'
