#Here are the steps I followed to getting ortools installed and running EXOSIMS
#I had no issues of any kind on my Ubuntu 16.04 installation.
#following instructions from https://developers.google.com/optimization/introduction/installing/binary

mkdir python-virtual-env
cd python-virtual-env
sudo pip install virtualenv
virtualenv 
virtualenv -p /usr/bin/python2.7 myVENV
source myVENV/bin/activate
cd myVENV
sudo apt-get install g++ make
pip install --upgrade ortools
